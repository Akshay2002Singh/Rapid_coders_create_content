def check_port(t_ip,port):
    try:
        sock = socket.socket()
        sock.settimeout(1)
        sock.connect((t_ip,port))
        print(colored(f"\t[+] Port {port} is open\n",'green'),end='')
    except:

def scan(temp_ip):
